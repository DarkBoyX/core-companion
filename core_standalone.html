Connecting Standalone to Roblox Studio
  1. Copy the Luau Script below.
  2. In Roblox Studio, create a script inside ServerScriptService.
  3. Paste the code, and ensure Allow HTTP Requests is enabled inside Game Settings.
  4. Run the place. Roblox Studio will communicate with the live host to receive dynamic Part generations in real-time.
Luau Script Source 
-- Core AI Roblox Studio Integration Standalone Bridge
local HttpService = game:GetService("HttpService")

local HOST_URL = "http://localhost:3000" -- Change this to your deployed App URL!
local POLL_INTERVAL = 1.5

while true do
    task.wait(POLL_INTERVAL)
    local success, result = pcall(function()
        return HttpService:GetAsync(HOST_URL .. "/api/roblox/poll")
    end)
    
    if success and result and result ~= "" then
        local data = HttpService:JSONDecode(result)
        if data and data.commands then
            for _, cmd in ipairs(data.commands) do
                if cmd.type == "create_part" then
                    local part = Instance.new("Part", workspace)
                    part.Name = cmd.name or "CorePart"
                    part.Size = Vector3.new(cmd.sizeX or 4, cmd.sizeY or 1, cmd.sizeZ or 4)
                    part.Position = Vector3.new(cmd.posX or 0, cmd.posY or 10, cmd.posZ or 0)
                    part.Anchored = true
                    print("[Core AI] Standalone Part Generated!")
                end
            end
        end
    end
end
